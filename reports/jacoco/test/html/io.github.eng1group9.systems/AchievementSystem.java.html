package io.github.eng1group9.systems;
import java.io.*;
import java.util.*;

public class AchievementSystem {

    private static final String FILE_NAME = "achievements.csv";

    private final List<Achievement> achievements = new ArrayList<>();

    // All the achievement names are below
    private String[] names = {"Find LONGBOI",
                              "Freeze the university",
                              "Achieve a score of 200000 or higher",
                              "Achieve a score of 400000 or higher",
                              "Escape with more than 4 minutes remaining",
                              "Beat the game!",
                              "Lose the game: This game is too hard :(",
                                "Get drunk?!"};

    public AchievementSystem() {
        loadAchievements();
    }

    public List<Achievement> getAchievements() {
        return new ArrayList<>(achievements);
    }
    /**
     * Loads the csv file on launch
     */
    private void loadAchievements() {
        File file = new File(FILE_NAME);
        if (!file.exists()) {
            try (BufferedWriter writer = new BufferedWriter(new FileWriter(FILE_NAME))) {
                writer.write("Achievement Name, Has been achieved");
                writer.newLine();

                for (String name : names) {
                    writer.write(name + "," + "false");
                    writer.newLine();

                    achievements.add(new Achievement(name, false));
                }

                System.out.println("Created new achievement save file at: " + file.getAbsolutePath());
            } catch (IOException e) {
                e.printStackTrace();
            }
            return;
        }

        try (BufferedReader reader = new BufferedReader(new FileReader(FILE_NAME))) {
            String line;
            reader.readLine(); // Skip the header line

            // reads each line and correctly adds the information to achievements
            while ((line = reader.readLine()) != null) {
                String[] parts = line.split(",");
                if (parts.length == 2) {
                    achievements.add(new Achievement(parts[0], Boolean.parseBoolean(parts[1])));
                }
            }

        } catch (IOException | NumberFormatException e) {
            e.printStackTrace();
        }
    }

    /**
     * Saves a change to the achievements in the csv file
     */
    private void saveAchievements() {
        try (BufferedWriter writer = new BufferedWriter(new FileWriter(FILE_NAME))) {

            writer.write("Achievement Name,Has been achieved");
            writer.newLine();

            for (Achievement a : achievements) {
                writer.write(a.getAchievementName() + "," + a.isAchieved());
                writer.newLine();
            }

        } catch (IOException e) {
            e.printStackTrace();
        }
    }

    /**
     * Used to when an achievement is achieved by the player
     *
     * @param name: Name of the achievement see "names" above
     *
     * @return boolean: whether the achievement status was updated
     * -> note: if the achievement was already achieved the function returns false, this boolean is used when displaying
     *          new achievements
     */
    public boolean achieveThis(String name) {

        boolean updated = false;

        for (Achievement a : achievements) {
            if (a.getAchievementName().equals(name)) {
                if (a.isAchieved()) {
                    System.out.println(name + " was already achieved");
                    return false;
                }

                a.setAchieved();
                System.out.println(name + " was updated to achieved");
                updated = true;
                break;
            }
        }

        if (updated) {
            saveAchievements();
            return true;
        }

        return false;
    }

    /**
     * Used to keep achievement names and whether they are achieved in one object to make things easier
     */
    public static class Achievement {
        private final String name;
        private boolean achieved;

        public Achievement(String name, boolean achieved) {
            this.name = name;
            this.achieved = achieved;
        }

        public String getAchievementName() {
            return name;
        }

        public boolean isAchieved() {
            return achieved;
        }

        public void setAchieved() {
            this.achieved = true;
        }
    }

}
