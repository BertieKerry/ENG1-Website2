package io.github.eng1group9.systems;
import java.io.*;
import java.util.*;


public class ScoreboardSystem {

    private static final int NUMBER_OF_SCORES = 5;
    private static final String FILE_NAME = "scoreboard.csv";

    private final List<Score> scores = new ArrayList<>();

    public ScoreboardSystem() {
        loadFromFile();
        System.out.println("Scoreboard loaded. File should be at: " + new File(FILE_NAME).getAbsolutePath());
    }

    /** Adds a score to the scoreboard and then sorts the scores and removes the lowest one from the board, saving the change
    to the csv file once done

     @param playerName: the current player's name
     @param score: the score the player achieved
     */
    public void addScore(String playerName, int score) {
        scores.add(new Score(playerName, score));
        sortScores();
        saveToFile();
    }

    /**
     * Gets the top scores from the csv file
     *
     * @return List<Score>: the top scores
     */
    public List<Score> getTopScores() {
        return new ArrayList<>(scores);
    }

    private void saveToFile() {
        try (BufferedWriter writer = new BufferedWriter(new FileWriter(FILE_NAME))) {
            writer.write("Player,Score");
            writer.newLine();

            for (Score s : scores) {
                writer.write(s.playerName + "," + s.score);
                writer.newLine();
            }
        } catch (IOException e) {
            e.printStackTrace();
        }
    }

    /**
     * Loads the csv file or creates a new file if one does not already exist
     */
    private void loadFromFile() {
        File file = new File(FILE_NAME);
        if (!file.exists()) {
            try (BufferedWriter writer = new BufferedWriter(new FileWriter(FILE_NAME))) {
                writer.write("Player,Score");
                writer.newLine();
                System.out.println("Created new scores file at: " + file.getAbsolutePath());
            } catch (IOException e) {
                e.printStackTrace();
            }
            return;
        }

        try (BufferedReader reader = new BufferedReader(new FileReader(FILE_NAME))) {
            String line;
            reader.readLine(); // Skip the header line

            // reads each line and correctly adds the information to scores
            while ((line = reader.readLine()) != null) {
                String[] parts = line.split(",");
                if (parts.length == 2) {
                    scores.add(new Score(parts[0], Integer.parseInt(parts[1])));
                }
            }

            sortScores();

        } catch (IOException | NumberFormatException e) {
            e.printStackTrace();
        }
    }

    /**
     * Sorts and orders the scores in descending order for displaying purposes
     */
    private void sortScores() {
        // Sorts scores in descending order
        scores.sort((a,b) -> Integer.compare(b.score, a.score)); // can swap b and a for ascending

        // Only gets the top n scores based on the defined NUMBER_OF_SCORES
        if (scores.size() > NUMBER_OF_SCORES) {
            scores.subList(NUMBER_OF_SCORES, scores.size()).clear(); // gets rid of any scores beyond the designated number of elements
        }
    }

    // Stores the players name and score in a single object for ease of use
    public static class Score {
        private final String playerName;
        private final int score;

        public Score(String playerName, int score) {
            this.playerName = playerName;
            this.score = score;
        }

        public String getPlayerName() {
            return playerName;
        }

        public int getScore() {
            return score;
        }
    }
}
